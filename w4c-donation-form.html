<script>

/* Pull your Squarespace button color to theme the card */

(function(){

  const wrap = document.currentScript.closest('.ea-card');

  if(!wrap) return;


  // Try CSS variables first, then a real button

  const root = getComputedStyle(document.documentElement);

  const vars = ['--button-background-color','--buttons-backgroundColor','--primary-button-background-color','--buttonFillColor','--theme-primary-color'];

  const cssVar = vars.map(v => root.getPropertyValue(v).trim()).find(Boolean);


  const probe = document.querySelector('.sqs-button-element--primary') || wrap.querySelector('.theme-proxy');

  const probeColor = probe ? (getComputedStyle(probe).backgroundColor || getComputedStyle(probe).borderTopColor) : '';


  const accent = cssVar || probeColor;

  function darken(rgb){ try{const m=rgb.match(/\d+/g).map(Number); return `rgb(${Math.max(0,m[0]-20)},${Math.max(0,m[1]-20)},${Math.max(0,m[2]-20)})`; }catch(e){ return null; } }

  function tint10(rgb){ try{const m=rgb.match(/\d+/g).map(Number); return `rgba(${m[0]},${m[1]},${m[2]},0.18)`; }catch(e){ return null; } }


  if(accent){

    wrap.style.setProperty('--accent', accent);

    wrap.style.setProperty('--accent-hover', darken(accent) || 'currentColor');

    wrap.style.setProperty('--accent-muted', tint10(accent) || 'rgba(0,0,0,.08)');

  }

})();

</script>
